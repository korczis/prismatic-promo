Prismatic Resilience

Chaos engineering and fault tolerance with circuit breakers and self-healing

260 files Infrastructure

Overview

Prismatic Resilience provides the platform's fault tolerance and chaos engineering infrastructure. In a system integrating 121+ external OSINT sources, government registries, and third-party APIs, failures are not exceptional events -- they are guaranteed. Prismatic Resilience ensures the platform degrades gracefully, recovers automatically, and maintains service quality even when individual components fail.

The application implements circuit breakers, bulkhead isolation, retry strategies, fallback chains, and chaos engineering tools that proactively test the platform's failure handling under controlled conditions.

Key Features

Circuit Breakers

# Define a circuit breaker for an external service
defmodule PrismaticResilience.Breakers.ShodanBreaker do
  use PrismaticResilience.CircuitBreaker,
    name: :shodan,
    failure_threshold: 5,        # Open after 5 failures
    reset_timeout: 60_000,       # Try again after 60 seconds
    half_open_max_calls: 2,      # Allow 2 test calls when half-open
    monitor_window: 120_000      # Track failures over 2 minutes
end

# Use circuit breaker in application code
case PrismaticResilience.call(:shodan, fn -> Shodan.host("1.2.3.4") end) do
  {:ok, result} ->
    process_result(result)

  {:error, :circuit_open} ->
    # Shodan circuit is open, use cached data or fallback
    PrismaticCache.get("shodan:host:1.2.3.4")

  {:error, reason} ->
    handle_error(reason)
end

# Check circuit breaker state
{:ok, state} = PrismaticResilience.breaker_state(:shodan)
# => %{state: :closed, failure_count: 1, last_failure: ~U[...]}

Fallback Chains

When a primary source fails, the system automatically falls through to alternatives.

# Define a fallback chain for host intelligence
{:ok, result} = PrismaticResilience.with_fallback([
  fn -> Shodan.host(ip) end,          # Primary: Shodan
  fn -> Censys.get_host(ip) end,      # Fallback 1: Censys
  fn -> PrismaticCache.get(cache_key) end,  # Fallback 2: Cached data
  fn -> {:ok, %{status: :partial, source: :none}} end  # Last resort: partial result
])

Bulkhead Isolation

# Isolate external service calls to prevent cascade failures
defmodule PrismaticResilience.Bulkheads do
  # Each bulkhead has its own process pool
  bulkhead :osint_queries, max_concurrent: 20, max_queue: 50
  bulkhead :czech_registries, max_concurrent: 5, max_queue: 10
  bulkhead :sanctions_screening, max_concurrent: 10, max_queue: 30

  def query_osint(source, query) do
    PrismaticResilience.execute_in_bulkhead(:osint_queries, fn ->
      apply(source, :search, [query])
    end)
  end
end

Retry Strategies

# Exponential backoff with jitter
{:ok, result} = PrismaticResilience.retry(
  fn -> ExternalApi.call(params) end,
  strategy: :exponential_backoff,
  base_delay: 1_000,          # 1 second initial delay
  max_delay: 30_000,          # Cap at 30 seconds
  max_retries: 5,
  jitter: :full,              # Add randomization to prevent thundering herd
  retry_on: [TimeoutError, ConnectionError],
  on_retry: fn attempt, delay ->
    Logger.warning("Retry attempt #{attempt}, waiting #{delay}ms")
  end
)

# Linear backoff for rate-limited APIs
{:ok, result} = PrismaticResilience.retry(
  fn -> RateLimitedApi.call(params) end,
  strategy: :linear_backoff,
  base_delay: 60_000,         # Wait 60 seconds between retries
  max_retries: 3,
  retry_on: [{:error, :rate_limited}]
)

Chaos Engineering

Proactively test failure handling by injecting controlled faults.

# Inject latency into Shodan calls (testing only)
PrismaticResilience.Chaos.inject(:shodan, :latency,
  delay: 5_000,               # Add 5 second delay
  probability: 0.3,           # 30% of calls
  duration: :timer.minutes(10)
)

# Inject failures into a specific service
PrismaticResilience.Chaos.inject(:censys, :failure,
  error: {:error, :connection_refused},
  probability: 0.5,
  duration: :timer.minutes(5)
)

# Run a chaos experiment
{:ok, report} = PrismaticResilience.Chaos.run_experiment(%{
  name: "osint_source_failure",
  description: "Test behavior when primary OSINT source is unavailable",
  faults: [
    %{target: :shodan, type: :failure, probability: 1.0},
    %{target: :censys, type: :latency, delay: 10_000}
  ],
  duration: :timer.minutes(15),
  assertions: [
    {:response_time, :p99, :<, 5_000},
    {:error_rate, :<, 0.01},
    {:fallback_activation, :>=, 1}
  ]
})
