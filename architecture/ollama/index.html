Local AI (Ollama)

Air-gapped LLM inference with qwen3-coder, deepseek

Overview

Prismatic integrates with Ollama for local, air-gapped LLM inference. This enables AI-powered features without sending data to external APIs - critical for security-sensitive intelligence operations.

# Use with Claude Code CLI
claude --model qwen3-coder "Generate REST API endpoint"

Elixir Integration

defmodule Prismatic.AI.Ollama do
  @moduledoc """
  Ollama LLM integration for local inference.
  """

  @base_url "http://localhost:11434"

  def generate(prompt, opts \\ []) do
    model = Keyword.get(opts, :model, "qwen3-coder")

    {:ok, response} =
      Req.post("#{@base_url}/api/generate", json: %{
        model: model,
        prompt: prompt,
        stream: false
      })

    {:ok, response.body["response"]}
  end

  def chat(messages, opts \\ []) do
    model = Keyword.get(opts, :model, "qwen3-coder")

    {:ok, response} =
      Req.post("#{@base_url}/api/chat", json: %{
        model: model,
        messages: messages,
        stream: false
      })

    {:ok, response.body["message"]["content"]}
  end
end

Optimization

# Connection pooling for Ollama requests
defmodule Prismatic.AI.Pool do
  use Supervisor

  def start_link(opts) do
    Supervisor.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def init(_opts) do
    children = [
      {Finch, name: OllamaFinch, pools: %{
        default: [size: 10, count: 2]
      }}
    ]

    Supervisor.init(children, strategy: :one_for_one)
  end
end

Cloud Fallback

Auto-fallback to cloud API when local Ollama is unavailable:

def generate_with_fallback(prompt, opts \\ []) do
  case Prismatic.AI.Ollama.generate(prompt, opts) do
    {:ok, response} ->
      {:ok, response, :local}

    {:error, :connection_refused} ->
      # Fallback to Anthropic API
      Logger.warning("Ollama unavailable, falling back to cloud")
      {:ok, response} = Prismatic.AI.Anthropic.generate(prompt, opts)
      {:ok, response, :cloud}
  end
end
