GraphQL API

Type-safe API with Absinthe, DataLoader, and real-time subscriptions

Overview

Prismatic Platform exposes a GraphQL API via Absinthe. It provides type-safe queries, mutations, and real-time subscriptions for all platform data, with automatic documentation and introspection.

Schema Architecture

GraphQL Request
      │
      ↓
  Router (/api/graphql)
      │
      ↓
  Absinthe.Plug
      │
      ↓
  Schema → Types → Resolvers → DataLoader → Database
      │                                         │
      ↓                                         ↓
  Response ←──────── Serialization ←────────── Data

Schema Definition

Root Types

defmodule PrismaticGraphQL.Schema do
  use Absinthe.Schema
  import_types(PrismaticGraphQL.Types.Asset)
  import_types(PrismaticGraphQL.Types.Agent)
  import_types(PrismaticGraphQL.Types.Security)

  query do
    @desc "Get all assets"
    field :assets, list_of(:asset) do
      arg(:filter, :asset_filter)
      arg(:limit, :integer, default_value: 20)
      arg(:offset, :integer, default_value: 0)
      resolve(&Resolvers.Asset.list/3)
    end

    @desc "Get asset by ID"
    field :asset, :asset do
      arg(:id, non_null(:id))
      resolve(&Resolvers.Asset.get/3)
    end
  end

  mutation do
    @desc "Discover assets for a domain"
    field :discover_assets, :discovery_result do
      arg(:domain, non_null(:string))
      resolve(&Resolvers.Asset.discover/3)
    end
  end

  subscription do
    @desc "Subscribe to asset discovery events"
    field :asset_discovered, :asset do
      config(fn _args, _info ->
        {:ok, topic: "assets:discovered"}
      end)
    end
  end
end

Object Types

defmodule PrismaticGraphQL.Types.Asset do
  use Absinthe.Schema.Notation
  import Absinthe.Resolution.Helpers, only: [dataloader: 1]

  object :asset do
    field :id, non_null(:id)
    field :domain, non_null(:string)
    field :ip_addresses, list_of(:string)
    field :discovered_at, :datetime
    field :security_rating, :security_rating, resolve: dataloader(:ratings)
    field :vulnerabilities, list_of(:vulnerability), resolve: dataloader(:vulns)
  end

  object :security_rating do
    field :grade, :string
    field :score, :integer
    field :industry_percentile, :integer
  end

  input_object :asset_filter do
    field :domain_contains, :string
    field :security_grade, :string
    field :discovered_after, :datetime
  end
end

DataLoader

Batched Data Loading

defmodule PrismaticGraphQL.DataLoader do
  def data() do
    Dataloader.new()
    |> Dataloader.add_source(:ratings, create_ratings_source())
    |> Dataloader.add_source(:vulns, create_vulns_source())
  end

  defp create_ratings_source() do
    Dataloader.Ecto.new(PrismaticStorage.Repo,
      query: fn queryable, _params ->
        from r in queryable, order_by: [desc: r.calculated_at]
      end
    )
  end
end

N+1 Prevention

# Without DataLoader: N+1 queries
# assets: 1 query
# ratings: N queries (one per asset)

# With DataLoader: 2 queries total
# assets: 1 query
# ratings: 1 batched query for all assets

Resolvers

Query Resolver

defmodule PrismaticGraphQL.Resolvers.Asset do
  def list(_parent, args, _resolution) do
    assets =
      Asset
      |> apply_filters(args[:filter])
      |> limit(^args[:limit])
      |> offset(^args[:offset])
      |> Repo.all()

    {:ok, assets}
  end

  def get(_parent, %{id: id}, _resolution) do
    case Repo.get(Asset, id) do
      nil -> {:error, "Asset not found"}
      asset -> {:ok, asset}
    end
  end
end

Mutation Resolver

defmodule PrismaticGraphQL.Resolvers.Asset do
  def discover(_parent, %{domain: domain}, %{context: context}) do
    with :ok <- authorize(context.current_user, :discover),
         {:ok, result} <- PrismaticPerimeter.discover(domain) do
      {:ok, result}
    end
  end
end

Authentication

Context Plug

defmodule PrismaticGraphQL.Context do
  def call(conn, _opts) do
    context = build_context(conn)
    Absinthe.Plug.put_options(conn, context: context)
  end

  defp build_context(conn) do
    with ["Bearer " <> token] <- get_req_header(conn, "authorization"),
         {:ok, user} <- authenticate(token) do
      %{current_user: user}
    else
      _ -> %{}
    end
  end
end

Authorization in Resolvers

def discover(_parent, args, %{context: %{current_user: user}}) do
  if Permissions.can?(user, :discover_assets) do
    PrismaticPerimeter.discover(args.domain)
  else
    {:error, "Unauthorized"}
  end
end

def discover(_parent, _args, _resolution) do
  {:error, "Authentication required"}
end
